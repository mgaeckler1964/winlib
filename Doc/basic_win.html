Methoden

bool create( const BASIC_WIN *parent, const char *className, ... );

Erzeugt ein neues Fenster. Liefert false zurück, wenn das Fenster erfolgreich erzeugt wurde.

bool BASIC_WIN::create( const BASIC_WIN *parent,
        const char *className, const char *winName=NULL,
        int x=CW_USEDEFAULT, int y=CW_USEDEFAULT,
        int nWidth=CW_USEDEFAULT, int nHeight=CW_USEDEFAULT,
        DWORD dwStyle = 0, HMENU menu=0, LPSTR lpParam=NULL )
{
        winHandle = CreateWindow( (LPSTR)className, (LPSTR)winName, dwStyle,
                x, y, nWidth, nHeight,
                parent ? parent->winHandle : 0,
                menu, instance, lpParam );
        return winHandle ? false : true;
}

virtual void close( void );

Schließt das Fenster.

void BASIC_WIN::close( void )
{
	if( parent )
		parent->handleChildClose( this );
        DestroyWindow( winHandle );
        winHandle = 0;
}
