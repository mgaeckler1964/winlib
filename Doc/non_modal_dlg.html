Erzeugt eine neue Dialogbox aus einer Ressource. Liefert false zurück, wenn der Dialog erfolgreich erzeugt wurde.

bool NON_MODAL_DLG::create( const char *dialog, const BASIC_WIN *parent )
{
	HWND	handle;

	registerWindow();
	handle = CreateDialog( instance, (LPSTR)dialog,
		parent ? parent->handle() : 0,
		(DLGPROC)WindowProc );
	if( handle )
		return false;
	else
	{
		unregisterWindow();
		return true;
	}
}

bool NON_MODAL_DLG::create( int dialog, const BASIC_WIN *parent )
{
	return create( (const char *)MAKEINTRESOURCE(dialog), parent );
}
